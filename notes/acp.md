# x-y

'x without y'

- x-x = 0
- a-b = a
- 0-x = 0
- a-0 = a
- a-1 = a
- 1-a = 1
- (x+y)-z = x-z + y-z
- (x+y) - y = x - y
- x - (y+z) = x-y-z
- ax-by = ax
- ax - ay = x-y?ax


# @

'all behaviour'

- @ x = @
- @+x = @
- x-@ = 0

# x?y

'if x=0, then x else y'

- 0?x = 0
- a?x = x
- ax ? y = y
- (x + y)? z = (x?y)?z
